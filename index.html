Routing


var app = angular.module('NgESS', ['NgESSFilters','ngRoute']);

app.config(['$routeProvider',
  function($routeProvider) {
    $routeProvider.
      when('/ng-repeat', {
        templateUrl: 'partials/ngrepeat.html',
        controller: 'orderCtrl'
      }).
	  when('/filter', {
        templateUrl: 'partials/filter.html',
        controller: 'orderCtrl'
      }).
      otherwise({
        redirectTo: '/'
      });
  }]);
